jar包更改配置

//定义jarfile为jar包名
jarfile="pushData8083.jar"

<!-- 生成temp临时目录 -->
unzip -x "$jarfile" -d "${jarfile}.tmp"  

<!-- 进入临时目录 -->
cd "${jarfile}.tmp"

<!-- 在临时目录更新文件 -->
zip -ru "../$jarfile" * 

<!-- 删除临时目录 -->
cd ../
rm -rf "${jarfile}.tmp"
